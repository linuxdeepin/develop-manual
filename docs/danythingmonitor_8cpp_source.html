134 void DAnythingMonitor::workSignal()
135 {
136 
137  std::lock_guard<std::mutex> raiiLock{ m_mutex };
138 
139  int fd = open(PROCFS_PATH, O_RDONLY);
140  if (fd < 0) {
141  notify();
142 
143  return;
144  }
145 
146  ioctl_rs_args irsa;
147  if (ioctl(fd, VC_IOCTL_READSTAT, &irsa) != 0) {
148  close(fd);
149  notify();
150 
151  return;
152  }
153 
154  if (irsa.cur_changes == 0) {
155  close(fd);
156  notify();
157 
158  return;
159  }
160 
161  char buf[1 << 20] {};
162  ioctl_rd_args ira ;
163  ira.data = buf;
164 
165 
166  while (true) {
167  ira.size = sizeof(buf);
168  if (ioctl(fd, VC_IOCTL_READDATA, &ira) != 0) {
169  break;
170  }
171 
172  // no more changes
173  if (ira.size == 0) {
174  break;
175  }
176 
177  int off = 0;
178  for (int i = 0; i < ira.size; i++) {
179  unsigned char action = static_cast<unsigned char>(*(ira.data + off));
180  off++;
181  char *src = ira.data + off, *dst = nullptr;
182  off += strlen(src) + 1;
183 
184  switch (action) {
186  // case ACT_NEW_FILE:
187  // case ACT_NEW_SYMLINK:
188  // case ACT_NEW_LINK:
189  // case ACT_NEW_FOLDER:
190  // {
191  // qDebug()<< act_names[action] << "-------->" << src;
192  // }
193  case ACT_DEL_FILE:
194  case ACT_DEL_FOLDER: {
195  m_changedFiles.emplace_back(QString{}, QString{src});
196  //#ifdef QT_DEBUG
197  // qDebug()<< act_names[action] << "--------->" << src;
198  //#endif
199  break;
200  }
201  case ACT_RENAME_FILE:
202  case ACT_RENAME_FOLDER: {
203  dst = ira.data + off;
204  off += strlen(dst) + 1;
205  m_changedFiles.emplace_back(QString{src},
206  QString{dst});
207  //#ifdef QT_DEBUG
208  // qDebug()<< act_names[action] << src << "--------->" << dst;
209  //#endif
210  break;
211  }
212  default:

214  }
215  }
