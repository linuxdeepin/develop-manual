175 static QString formatCap(qulonglong cap, const int size = 1024, quint8 precision = 1)
176 {
177  static QString type[] = {" B", " KB", " MB", " GB", " TB"};
178 
179  qulonglong lc = cap;
180  double dc = cap;
181  double ds = size;
182 
183  for (size_t p = 0; p < sizeof(type); ++p) {
184  if (cap < pow(size, p + 1) || p == sizeof(type) - 1) {
185  if (!precision) {
186  return QString::number(round(lc / pow(size, p))) + type[p];
187  }
188 
189  return QString::number(dc / pow(ds, p), 'f', precision) + type[p];
190  }
191  }
192 
193  return "";
194 }

196 QHash<QString, QString> ComputerPropertyDialog::getMessage(const QStringList &data)
197 {
198  QHash<QString, QString> datas;
199  QString Edition;
200  QString version;
201  QString memoryInstallStr;
202  QString memoryStr;
203  QString processor;
204  QString systemType;
205 
207  QDBusInterface *deepin_systemInfo = new QDBusInterface("com.deepin.system.SystemInfo",
208  "/com/deepin/system/SystemInfo",
209  "com.deepin.system.SystemInfo",
210  QDBusConnection::systemBus(), this);
211 
212  //部分数据优先从dbus读取，如果dbus没有，则从dtk读数据
213  if (m_systemInfo->isValid()) {
215  memoryInstallStr = formatCap(deepin_systemInfo->property("MemorySize").toULongLong(), 1024, 0);
217  memoryStr = formatCap(static_cast<qulonglong>(DSysInfo::memoryTotalSize()));
219  processor = m_systemInfo->processor();
221  systemType = QString::number(m_systemInfo->systemType()) + tr("Bit");
222 
223  if (Edition.isEmpty()) {
224  Edition = m_systemInfo->version();
225  QStringList temp = Edition.split(' ');
226  if (temp.size() > 1) {
227  version = temp[0];
228  Edition = temp[1];
229  } else if (!temp.isEmpty()) {
230  Edition = temp[0];
231  }
232  }
233  }
234 
235  if (DSysInfo::isDeepin()) {
237  Edition = DSysInfo::uosEditionName();
239  version = DSysInfo::majorVersion();
240  if (!deepin_systemInfo->isValid()) {
242  memoryInstallStr = formatCap(static_cast<qulonglong>(DSysInfo::memoryInstalledSize()), 1024, 0);
244  memoryStr = formatCap(static_cast<qulonglong>(DSysInfo::memoryTotalSize()));
246  processor = DSysInfo::cpuModelName();
247  }
248  }
249 
250  datas.insert(data.at(0), DSysInfo::computerName());
251  datas.insert(data.at(1), Edition);
252  datas.insert(data.at(2), version);
253  datas.insert(data.at(3), systemType);
254  datas.insert(data.at(4), processor);
255  datas.insert(data.at(5), memoryInstallStr + "(" + memoryStr + ' ' + tr("Available") + ")");
256 
257  return datas;
258 }
