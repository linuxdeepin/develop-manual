55 // Parses a string as a command line flag. The string should have the
56 // format "--gmock_flag=value". When def_optional is true, the
57 // "=value" part can be omitted.
58 //
59 // Returns the value of the flag, or NULL if the parsing failed.
60 static const char *ParseGoogleMockFlagValue(const char *str,
61  const char *flag,
62  bool def_optional)
63 {
64  // str and flag must not be NULL.
65  if (str == nullptr || flag == nullptr)
66  return nullptr;
67 
68  // The flag must start with "--gmock_".
69  const std::string flag_str = std::string("--gmock_") + flag;
70  const size_t flag_len = flag_str.length();
71  if (strncmp(str, flag_str.c_str(), flag_len) != 0)
72  return nullptr;
73 
74  // Skips the flag name.
75  const char *flag_end = str + flag_len;
76 
77  // When def_optional is true, it's OK to not have a "=value" part.
78  if (def_optional && (flag_end[0] == '\0')) {
79  return flag_end;
80  }
81 
82  // If def_optional is true and there are more characters after the
83  // flag name, or if def_optional is false, there must be a '=' after
84  // the flag name.
85  if (flag_end[0] != '=')
86  return nullptr;
87 
88  // Returns the string after "=".
89  return flag_end + 1;
90 }

92 // Parses a string for a Google Mock bool flag, in the form of
93 // "--gmock_flag=value".
94 //
95 // On success, stores the value of the flag in *value, and returns
96 // true. On failure, returns false without changing *value.
97 static bool ParseGoogleMockBoolFlag(const char *str, const char *flag,
98  bool *value)
99 {
100  // Gets the value of the flag as a string.
101  const char *const value_str = ParseGoogleMockFlagValue(str, flag, true);
102 
103  // Aborts if the parsing failed.
104  if (value_str == nullptr)
105  return false;
106 
107  // Converts the string value to a bool.
108  *value = !(*value_str == '0' || *value_str == 'f' || *value_str == 'F');
109  return true;
110 }

112 // Parses a string for a Google Mock string flag, in the form of
113 // "--gmock_flag=value".
114 //
115 // On success, stores the value of the flag in *value, and returns
116 // true. On failure, returns false without changing *value.
117 template<typename String>
118 static bool ParseGoogleMockStringFlag(const char *str, const char *flag,
119  String *value)
120 {
121  // Gets the value of the flag as a string.
122  const char *const value_str = ParseGoogleMockFlagValue(str, flag, false);
123 
124  // Aborts if the parsing failed.
125  if (value_str == nullptr)
126  return false;
127 
128  // Sets *value to the value of the flag.
129  *value = value_str;
130  return true;
131 }
132 
133 static bool ParseGoogleMockIntFlag(const char *str, const char *flag,
134  int *value)
135 {
136  // Gets the value of the flag as a string.
137  const char *const value_str = ParseGoogleMockFlagValue(str, flag, true);
138 
139  // Aborts if the parsing failed.
140  if (value_str == nullptr)
141  return false;
142 
143  // Sets *value to the value of the flag.
144  return ParseInt32(Message() << "The value of flag --" << flag,
145  value_str, value);
146 }

148 // The internal implementation of InitGoogleMock().
149 //
150 // The type parameter CharType can be instantiated to either char or
151 // wchar_t.
152 template<typename CharType>
153 void InitGoogleMockImpl(int *argc, CharType **argv)
154 {
155  // Makes sure Google Test is initialized. InitGoogleTest() is
156  // idempotent, so it's fine if the user has already called it.
157  InitGoogleTest(argc, argv);
158  if (*argc <= 0)
159  return;
160 
161  for (int i = 1; i != *argc; i++) {
162  const std::string arg_string = StreamableToString(argv[i]);
163  const char *const arg = arg_string.c_str();
164 
165  // Do we see a Google Mock flag?
166  if (ParseGoogleMockBoolFlag(arg, "catch_leaked_mocks",
167  &GMOCK_FLAG(catch_leaked_mocks))
168  || ParseGoogleMockStringFlag(arg, "verbose", &GMOCK_FLAG(verbose)) || ParseGoogleMockIntFlag(arg, "default_mock_behavior", &GMOCK_FLAG(default_mock_behavior))) {
169  // Yes. Shift the remainder of the argv list left by one. Note
170  // that argv has (*argc + 1) elements, the last one always being
171  // NULL. The following loop moves the trailing NULL element as
172  // well.
173  for (int j = i; j != *argc; j++) {
174  argv[j] = argv[j + 1];
175  }
176 
177  // Decrements the argument count.
178  (*argc)--;
179 
180  // We also need to decrement the iterator as we just removed
181  // an element.
182  i--;
183  }
184  }
185 }

189 // Initializes Google Mock. This must be called before running the
190 // tests. In particular, it parses a command line for the flags that
191 // Google Mock recognizes. Whenever a Google Mock flag is seen, it is
192 // removed from argv, and *argc is decremented.
193 //
194 // No value is returned. Instead, the Google Mock flag variables are
195 // updated.
196 //
197 // Since Google Test is needed for Google Mock to work, this function
198 // also initializes Google Test and parses its flags, if that hasn't
199 // been done.
200 GTEST_API_ void InitGoogleMock(int *argc, char **argv)
201 {
202  internal::InitGoogleMockImpl(argc, argv);
203 }
